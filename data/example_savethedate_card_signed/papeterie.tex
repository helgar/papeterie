\documentclass[a5paper]{scrartcl}

% For German umlauts
\usepackage[ngerman]{babel}

% Needed to include pictures
\usepackage{graphicx}

% Specify a custom format. This is the biggest card that is allowed to be sent as
% 'normal' postcard by the German mail. Check the rules in your country before
% sending this.
\usepackage[paperwidth=175mm,
    paperheight=125mm,
    left=5mm,
    right=5mm,
    top=5mm,
    bottom=5mm]{geometry}

% Set default font to typewriter-like font.
\renewcommand{\familydefault}{\ttdefault}

% Using Courier font
\renewcommand{\ttdefault}{pcr}

% Do not indent first line of paragraphs.
\setlength{\parindent}{0pt}

% Convert environement so that we can use absolute-positioned text boxes.
\usepackage[absolute]{textpos}
\setlength{\TPHorizModule}{10mm}
\setlength{\TPVertModule}{\TPHorizModule}
\textblockorigin{0mm}{0mm}

% Stuff necessary to have a nice cursive font

\usepackage{calligra}
\usepackage[T1]{fontenc}

% Color of the mentioning of the source.
\usepackage{xcolor}
\definecolor{grey}{cmyk}{0, 0, 0, 0.5}

% For setstretch, custom line spacing
\usepackage{setspace}
